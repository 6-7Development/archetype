    <script>
        function showNotification(message) {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.style.display = 'block';
            setTimeout(() => {
                notification.style.display = 'none';
            }, 3000);
        }
        
        function navigateTo(page) {
            document.querySelectorAll('.nav-item').forEach(item => {
                item.classList.remove('active');
            });
            event.target.closest('.nav-item').classList.add('active');
            showNotification(`Navigating to ${page.replace('-', ' ').toUpperCase()}`);
        }
        
        function cardClick(type) {
            showNotification(`Opening ${type.replace('-', ' ').toUpperCase()} details`);
        }
        
        function quickAction(action) {
            showNotification(`${action.replace('-', ' ').toUpperCase()} initiated`);
        }
        
        function rowClick(project) {
            showNotification(`Loading ${project.replace('-', ' ')} project details`);
        }
        
        function actionClick(action, project) {
            showNotification(`${action.toUpperCase()}: ${project.replace('-', ' ')}`);
        }
    </script>
