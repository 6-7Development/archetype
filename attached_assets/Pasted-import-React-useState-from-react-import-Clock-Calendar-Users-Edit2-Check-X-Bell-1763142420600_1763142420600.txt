                    <Download className="w-4 h-4" />
                    <span>Export</span>
                  </button>
                )}
              </div>
            </div>

            {/* Timesheet Entries */}
            <div className="bg-white rounded-lg shadow-sm border border-gray-200 overflow-hidden">
              {/* Desktop Table */}
              <div className="hidden lg:block overflow-x-auto">
                <table className="w-full">
                  <thead className="bg-gray-50 border-b border-gray-200">
                    <tr>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Employee</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Date</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Clock In</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Clock Out</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Break</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Total</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Status</th>
                      <th className="px-4 py-3 text-left text-xs font-medium text-gray-600 uppercase">Actions</th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-gray-200">
                    {timeEntries.map(entry => (
                      <tr key={entry.id} className="hover:bg-gray-50">
                        <td className="px-4 py-3">
                          <div className="font-medium text-gray-900">{entry.employeeName}</div>
                          {entry.hasPhoto && (
                            <div className="flex items-center space-x-1 text-xs text-gray-500 mt-1">
                              <Camera className="w-3 h-3" />
                              <span>Photo verified</span>
                            </div>
                          )}
                        </td>
                        <td className="px-4 py-3 text-sm text-gray-900">{entry.date}</td>
                        <td className="px-4 py-3 text-sm text-gray-900">{entry.clockIn}</td>
                        <td className="px-4 py-3 text-sm text-gray-900">{entry.clockOut}</td>
                        <td className="px-4 py-3 text-sm text-gray-900">{entry.breakTime} min</td>
                        <td className="px-4 py-3">
                          <span className="font-bold text-gray-900">{entry.totalHours}h</span>
                        </td>
                        <td className="px-4 py-3">
                          <span className={`inline-flex items-center px-2 py-1 rounded-full text-xs font-medium ${
                            entry.status === 'approved' ? 'bg-green-100 text-green-700' :
                            entry.status === 'pending_approval' ? 'bg-yellow-100 text-yellow-700' :
                            'bg-red-100 text-red-700'
                          }`}>
                            {entry.status === 'approved' && <CheckCircle className="w-3 h-3 mr-1" />}
                            {entry.status === 'pending_approval' && <Clock className="w-3 h-3 mr-1" />}
                            {entry.status === 'rejected' && <XCircle className="w-3 h-3 mr-1" />}
                            {entry.status.replace('_', ' ')}
                          </span>
                        </td>
                        <td className="px-4 py-3">
                          <div className="flex items-center space-x-2">
                            <button 
                              onClick={() => {
                                setSelectedEntry(entry);
                                setShowEditModal(true);
                              }}
                              className="p-1 text-blue-600 hover:bg-blue-50 rounded"
                            >
                              <Eye className="w-4 h-4" />
                            </button>
                            {currentUser.permissions.editTeamTime && (
                              <button className="p-1 text-gray-600 hover:bg-gray-50 rounded">
                                <Edit2 className="w-4 h-4" />
                              </button>
                            )}
                          </div>
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>

              {/* Mobile Cards */}
              <div className="lg:hidden divide-y divide-gray-200">
                {timeEntries.map(entry => (
                  <div key={entry.id} className="p-4">
                    <div className="flex items-center justify-between mb-3">
                      <div>
                        <p className="font-bold text-gray-900">{entry.employeeName}</p>
                        <p className="text-sm text-gray-600">{entry.date}</p>
                      </div>
                      <span className={`px-2 py-1 rounded-full text-xs font-medium ${
                        entry.status === 'approved' ? 'bg-green-100 text-green-700' :
                        entry.status === 'pending_approval' ? 'bg-yellow-100 text-yellow-700' :
                        'bg-red-100 text-red-700'
                      }`}>
                        {entry.status.replace('_', ' ')}
                      </span>
                    </div>

                    <div className="grid grid-cols-2 gap-2 text-sm mb-3">
                      <div>
                        <p className="text-gray-600">Clock In</p>
                        <p className="font-medium text-gray-900">{entry.clockIn}</p>
                      </div>
                      <div>
                        <p className="text-gray-600">Clock Out</p>
                        <p className="font-medium text-gray-900">{entry.clockOut}</p>
                      </div>
                      <div>
                        <p className="text-gray-600">Break</p>
                        <p className="font-medium text-gray-900">{entry.breakTime} min</p>
                      </div>
                      <div>
                        <p className="text-gray-600">Total Hours</p>
                        <p className="font-bold text-gray-900">{entry.totalHours}h</p>
                      </div>
                    </div>

                    <button
                      onClick={() => {
                        setSelectedEntry(entry);
                        setShowEditModal(true);
                      }}
                      className="w-full bg-blue-600 text-white py-2 rounded-lg text-sm font-medium"
                    >
                      View Details
                    </button>
                  </div>
                ))}
              </div>
            </div>
          </div>
        )}

        {/* Approvals View */}
        {view === 'approvals' && currentUser.permissions.approveTime && (
          <div className="space-y-4">
            <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-4 lg:p-6">
              <h3 className="text-lg font-bold text-gray-900 mb-4">
                Pending Approvals ({timeEntries.filter(e => e.status === 'pending_approval').length})
              </h3>

              <div className="space-y-4">
                {timeEntries.filter(e => e.status === 'pending_approval').map(entry => (
                  <div key={entry.id} className="border-2 border-orange-200 rounded-lg p-4 bg-orange-50">
                    <div className="flex flex-col lg:flex-row lg:items-center lg:justify-between gap-4">
                      <div className="flex-1">
                        <div className="flex items-center space-x-3 mb-2">
                          <div className="w-10 h-10 bg-gradient-to-br from-blue-500 to-indigo-600 rounded-full flex items-center justify-center text-white font-bold">
                            {entry.employeeName.split(' ').map(n => n[0]).join('')}
                          </div>
                          <div>
                            <p className="font-bold text-gray-900">{entry.employeeName}</p>
                            <p className="text-sm text-gray-600">{entry.date}</p>
                          </div>
                        </div>

                        <div className="grid grid-cols-2 lg:grid-cols-4 gap-2 text-sm">
                          <div>
                            <p className="text-gray-600">In: {entry.clockIn}</p>
                          </div>
                          <div>
                            <p className="text-gray-600">Out: {entry.clockOut}</p>
                          </div>
                          <div>
                            <p className="text-gray-600">Break: {entry.breakTime}m</p>
                          </div>
                          <div>
                            <p className="font-bold text-gray-900">Total: {entry.totalHours}h</p>
                          </div>
                        </div>

                        {entry.notes && (
                          <p className="text-sm text-gray-600 mt-2 italic">{entry.notes}</p>
                        )}
                      </div>

                      <div className="flex gap-2">
                        <button
                          onClick={() => handleApproveEntry(entry.id)}
                          className="flex-1 lg:flex-none bg-green-600 text-white px-6 py-2 rounded-lg font-medium hover:bg-green-700 flex items-center justify-center space-x-2"
                        >
                          <CheckCircle className="w-4 h-4" />
                          <span>Approve</span>
                        </button>
                        <button
                          onClick={() => handleRejectEntry(entry.id)}
                          className="flex-1 lg:flex-none bg-red-600 text-white px-6 py-2 rounded-lg font-medium hover:bg-red-700 flex items-center justify-center space-x-2"
                        >
                          <XCircle className="w-4 h-4" />
                          <span>Reject</span>
                        </button>
                      </div>
                    </div>
                  </div>
                ))}

                {timeEntries.filter(e => e.status === 'pending_approval').length === 0 && (
                  <div className="text-center py-8">
                    <CheckCircle className="w-16 h-16 text-green-500 mx-auto mb-3" />
                    <p className="text-gray-600 font-medium">All caught up!</p>
                    <p className="text-sm text-gray-500">No pending approvals</p>
                  </div>
                )}
              </div>
            </div>
          </div>
        )}

        {/* Audit Log View */}
        {view === 'audit' && currentUser.permissions.viewAudit && (
          <div className="bg-white rounded-lg shadow-sm border border-gray-200 overflow-hidden">
            <div className="p-4 lg:p-6 border-b border-gray-200">
              <h3 className="text-lg font-bold text-gray-900">Audit Trail</h3>
              <p className="text-sm text-gray-600">Complete history of all timesheet actions</p>
            </div>

            <div className="divide-y divide-gray-200">
              {auditLog.map(log => (
                <div key={log.id} className="p-4 hover:bg-gray-50">
                  <div className="flex items-start space-x-3">
                    <div className={`p-2 rounded-lg ${
                      log.action.includes('approved') ? 'bg-green-100' :
                      log.action.includes('rejected') ? 'bg-red-100' :
                      log.action.includes('edited') ? 'bg-blue-100' :
                      'bg-gray-100'
                    }`}>
                      {log.action.includes('approved') && <CheckCircle className="w-5 h-5 text-green-600" />}
                      {log.action.includes('rejected') && <XCircle className="w-5 h-5 text-red-600" />}
                      {log.action.includes('edited') && <Edit2 className="w-5 h-5 text-blue-600" />}
                      {log.action.includes('clock') && <Clock className="w-5 h-5 text-gray-600" />}
                    </div>
                    <div className="flex-1 min-w-0">
                      <div className="flex items-center justify-between gap-2">
                        <p className="font-medium text-gray-900 truncate">
                          {log.userName} - {log.action.replace('_', ' ')}
                        </p>
                        <span className="text-xs text-gray-500 flex-shrink-0">{log.timestamp}</span>
                      </div>
                      <p className="text-sm text-gray-600 mt-1">{log.target}</p>
                      <p className="text-xs text-gray-500 mt-1">{log.details}</p>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}
      </div>

      {/* Mobile Bottom Navigation */}
      <div className="lg:hidden fixed bottom-0 left-0 right-0 bg-white border-t border-gray-200 z-30">
        <div className="grid grid-cols-4 h-16">
          <button
            onClick={() => setView('clock')}
            className={`flex flex-col items-center justify-center ${
              view === 'clock' ? 'text-blue-600' : 'text-gray-600'
            }`}
          >
            <Clock className="w-5 h-5 mb-1" />
            <span className="text-xs">Clock</span>
          </button>
          <button
            onClick={() => setView('timesheet')}
            className={`flex flex-col items-center justify-center ${
              view === 'timesheet' ? 'text-blue-600' : 'text-gray-600'
            }`}
          >
            <Calendar className="w-5 h-5 mb-1" />
            <span className="text-xs">Timesheet</span>
          </button>
          {currentUser.permissions.approveTime && (
            <button
              onClick={() => setView('approvals')}
              className={`flex flex-col items-center justify-center relative ${
                view === 'approvals' ? 'text-blue-600' : 'text-gray-600'
              }`}
            >
              <CheckCircle className="w-5 h-5 mb-1" />
              <span className="text-xs">Approve</span>
              {pendingApprovals > 0 && (
                <span className="absolute top-1 right-6 w-4 h-4 bg-orange-500 text-white text-xs rounded-full flex items-center justify-center font-bold">
                  {pendingApprovals}
                </span>
              )}
            </button>
          )}
          <button
            onClick={() => setView('audit')}
            className={`flex flex-col items-center justify-center ${
              view === 'audit' ? 'text-blue-600' : 'text-gray-600'
            }`}
          >
            <History className="w-5 h-5 mb-1" />
            <span className="text-xs">Audit</span>
          </button>
        </div>
      </div>

      {/* Notifications Panel */}
      {showNotifications && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 lg:flex lg:justify-end">
          <div 
            className="absolute inset-0 lg:relative"
            onClick={() => setShowNotifications(false)}
          ></div>
          <div className="absolute bottom-0 lg:relative lg:bottom-auto left-0 right-0 lg:left-auto lg:w-96 bg-white rounded-t-2xl lg:rounded-none lg:h-full shadow-2xl overflow-y-auto">
            <div className="sticky top-0 bg-white border-b border-gray-200 p-4 flex items-center justify-between">
              <h3 className="font-bold text-gray-900">Notifications</h3>
              <button onClick={() => setShowNotifications(false)} className="p-1">
                <X className="w-5 h-5" />
              </button>
            </div>

            <div className="divide-y divide-gray-200">
              {notifications.map(notif => (
                <div key={notif.id} className={`p-4 ${!notif.read ? 'bg-blue-50' : ''}`}>
                  <div className="flex items-start space-x-3">
                    <div className="p-2 rounded-lg bg-blue-100">
                      <Bell className="w-4 h-4 text-blue-600" />
                    </div>
                    <div className="flex-1">
                      <p className="text-sm text-gray-900 font-medium">{notif.message}</p>
                      <p className="text-xs text-gray-500 mt-1">{notif.timestamp}</p>
                    </div>
                    {!notif.read && (
                      <div className="w-2 h-2 bg-blue-600 rounded-full flex-shrink-0 mt-2"></div>
                    )}
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* Entry Details Modal */}
      {showEditModal && selectedEntry && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-center justify-center p-4">
          <div className="bg-white rounded-xl shadow-2xl max-w-2xl w-full max-h-[90vh] overflow-y-auto">
            <div className="sticky top-0 bg-gradient-to-r from-blue-600 to-indigo-600 text-white p-4 rounded-t-xl">
              <div className="flex items-center justify-between">
                <h3 className="text-lg font-bold">Timesheet Entry Details</h3>
                <button onClick={() => setShowEditModal(false)} className="p-1">
                  <X className="w-6 h-6" />
                </button>
              </div>
            </div>

            <div className="p-6 space-y-4">
              <div>
                <h4 className="font-bold text-gray-900 mb-2">{selectedEntry.employeeName}</h4>
                <p className="text-sm text-gray-600">{selectedEntry.date}</p>
              </div>

              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-1">Clock In</label>
                  <input
                    type="text"
                    value={selectedEntry.clockIn}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg"
                    disabled={!currentUser.permissions.editTeamTime}
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-1">Clock Out</label>
                  <input
                    type="text"
                    value={selectedEntry.clockOut}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg"
                    disabled={!currentUser.permissions.editTeamTime}
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-1">Break (min)</label>
                  <input
                    type="number"
                    value={selectedEntry.breakTime}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg"
                    disabled={!currentUser.permissions.editTeamTime}
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-1">Total Hours</label>
                  <input
                    type="text"
                    value={`${selectedEntry.totalHours}h`}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg bg-gray-50"
                    disabled
                  />
                </div>
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Location</label>
                <input
                  type="text"
                  value={selectedEntry.location}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg"
                  disabled
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Notes</label>
                <textarea
                  value={selectedEntry.notes}
                  rows={3}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg"
                  placeholder="Add notes..."
                  disabled={!currentUser.permissions.editTeamTime}
                />
              </div>

              {selectedEntry.modifications && selectedEntry.modifications.length > 0 && (
                <div>
                  <h5 className="font-medium text-gray-900 mb-2">Modification History</h5>
                  <div className="bg-gray-50 rounded-lg p-3 space-y-2">
                    {selectedEntry.modifications.map((mod, idx) => (
                      <div key={idx} className="text-sm">
                        <p className="text-gray-900">
                          <span className="font-medium">{mod.modifiedBy}</span> changed {mod.field}
                        </p>
                        <p className="text-gray-600">
                          From: {mod.oldValue} â†’ To: {mod.newValue}
                        </p>
                        <p className="text-xs text-gray-500">{mod.modifiedAt}</p>
                      </div>
                    ))}
                  </div>
                </div>
              )}

              {currentUser.permissions.approveTime && selectedEntry.status === 'pending_approval' && (
                <div className="flex gap-3 pt-4">
                  <button
                    onClick={() => {
                      handleApproveEntry(selectedEntry.id);
                      setShowEditModal(false);
                    }}
                    className="flex-1 bg-green-600 text-white py-3 rounded-lg font-medium hover:bg-green-700"
                  >
                    Approve
                  </button>
                  <button
                    onClick={() => {
                      handleRejectEntry(selectedEntry.id);
                      setShowEditModal(false);
                    }}
                    className="flex-1 bg-red-600 text-white py-3 rounded-lg font-medium hover:bg-red-700"
                  >
                    Reject
                  </button>
                </div>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default UniversalTimesheet;
